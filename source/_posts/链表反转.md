LeetCode 206. Reverse Linked List
<!--more-->

```cpp
struct ListNode {
	int val;
	ListNode* next;
	ListNode(int x) : val(x),next(nullptr) {}
};

ListNode* reverseSingleList(ListNode* head) {
	ListNode* pre = nullptr;
	while(head)
	{
		ListNode* temp = head->next;
		head->next = pre;
		pre = head;
		head = temp;
	}
	return pre;
}
```
