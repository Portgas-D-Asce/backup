LeetCode 160. Intersection of Two Linked Lists
<!--more-->

```cpp
struct ListNode {
	int val;
	ListNode* next;
	ListNode(int x) : val(x),next(NULL) {}
};

ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
	ListNode* pa = headA;
	ListNode* pb = headB;
	while (pa != pb)
	{
		pa = pa ? pa->next : headB;
		pb = pb ? pb->next : headA;
	}
	return pa;
}
```
