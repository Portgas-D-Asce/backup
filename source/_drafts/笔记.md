## 素数统计（筛法）
```c
//统计 num 以内（包括 num）素数的个数；
#include <stdio.h>
#include <stdlib.h>
#include <malloc.h>

int main() {
	const int num = 1000000;
    int *flag = (int *)calloc(num + 1, sizeof(int));
    for(int i = 2; i * i <= num; ++i) {
        if(flag[i] == 0) {
            int temp = num / i;
            for(int j = 2; j <= temp; ++j) {
                flag[i * j] = 1;
            }
        }
    }
    int cnt = 0;
    for(int i = 2; i <= num; ++i) {
        if(flag[i] == 0) {
            ++cnt;
        }
    }
	printf("%d\n", cnt);
    free(flag);
    flag == NULL;

	return 0;
}
```
